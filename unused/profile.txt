import React, { useState } from 'react';
import { View, Text, StyleSheet, ScrollView, TouchableOpacity } from 'react-native';
import { useAuth } from '../contexts/AuthContext';
import { Button } from '../components/ui/Button';
import { ConfirmDialog } from '../components/ui/ConfirmDialog';
import { router } from 'expo-router';

export default function ProfileScreen() {
  const { user, logout } = useAuth();
  const [showLogoutDialog, setShowLogoutDialog] = useState(false);
  const [isLoggingOut, setIsLoggingOut] = useState(false);

  const handleLogoutPress = () => {
    console.log('üîµ Logout button pressed');
    setShowLogoutDialog(true);
  };

  const handleLogoutConfirm = async () => {
    console.log('üü¢ Logout confirmed by user');
    setIsLoggingOut(true);
    
    try {
      console.log('üü° Calling logout function...');
      await logout();
      console.log('‚úÖ Logout successful, closing dialog');
      setShowLogoutDialog(false);
      
      // Nawigacja do ekranu logowania
      console.log('üîÑ Navigating to login screen...');
      router.replace('/(auth)/login');
      console.log('‚úÖ Navigation complete');
    } catch (error) {
      console.error('‚ùå Logout error:', error);
      setShowLogoutDialog(false);
      alert('WystƒÖpi≈Ç b≈ÇƒÖd podczas wylogowywania. Spr√≥buj ponownie.');
    } finally {
      setIsLoggingOut(false);
    }
  };

  const handleLogoutCancel = () => {
    console.log('üî¥ Logout cancelled by user');
    setShowLogoutDialog(false);
  };

  if (!user) {
    return (
      <View style={styles.container}>
        <Text style={styles.loadingText}>≈Åadowanie...</Text>
      </View>
    );
  }

  return (
    <>
      <ScrollView style={styles.container}>
        <View style={styles.content}>
          <View style={styles.header}>
            <View style={styles.avatarContainer}>
              <Text style={styles.avatarText}>
                {user.username.charAt(0).toUpperCase()}
              </Text>
            </View>
            <Text style={styles.username}>{user.username}</Text>
            <Text style={styles.email}>{user.email}</Text>
          </View>

          <View style={styles.section}>
            <Text style={styles.sectionTitle}>Informacje o koncie</Text>
            
            <View style={styles.infoRow}>
              <Text style={styles.infoLabel}>Nazwa u≈ºytkownika</Text>
              <Text style={styles.infoValue}>{user.username}</Text>
            </View>
            
            <View style={styles.infoRow}>
              <Text style={styles.infoLabel}>Email</Text>
              <Text style={styles.infoValue}>{user.email}</Text>
            </View>
            
            <View style={styles.infoRow}>
              <Text style={styles.infoLabel}>Imiƒô</Text>
              <Text style={styles.infoValue}>{user.first_name || 'Nie podano'}</Text>
            </View>
            
            <View style={styles.infoRow}>
              <Text style={styles.infoLabel}>Nazwisko</Text>
              <Text style={styles.infoValue}>{user.last_name || 'Nie podano'}</Text>
            </View>
          </View>

          <View style={styles.buttonContainer}>
            <Button
              title="Wyloguj siƒô"
              onPress={handleLogoutPress}
              variant="outline"
              style={styles.logoutButton}
            />
          </View>
        </View>
      </ScrollView>

      <ConfirmDialog
        visible={showLogoutDialog}
        title="Wylogowanie"
        message="Czy na pewno chcesz siƒô wylogowaƒá?"
        confirmText="Wyloguj"
        cancelText="Anuluj"
        onConfirm={handleLogoutConfirm}
        onCancel={handleLogoutCancel}
        confirmColor="#ef4444"
      />
    </>
  );
}
